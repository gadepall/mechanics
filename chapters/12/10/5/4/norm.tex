\section{\textbf{Problem}}
if $\vec{a}=\vec{b}+\vec{c}$, then is true that $|\vec{a}|=|\vec{b}|+|\vec{c}|$?
Justify your answer .
\section{\textbf{Solution}}
Given
\begin{align}
    \vec{a}=\vec{b}+\vec{c}
\end{align}
\fi
Let
\begin{align}
    \vec{b}=\myvec{1\\2\\3},\vec{c}=\myvec{2\\-1\\-2}
\end{align}
Then
\begin{align}
	\vec{a}&=\vec{b}+\vec{c}
    =\myvec{3\\1\\1}
    \\
	\implies
    \norm{\vec{a}}
    &=\sqrt{11},\,
    \norm{\vec{b}}    =\sqrt{14},\,
    \norm{\vec{c}}
    =3.
\end{align}
Thus
\begin{align}
	\norm{\vec{a}}\ne\norm{\vec{b}}+\norm{\vec{c}}
\end{align}
