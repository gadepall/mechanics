Let the $x$ intercept be $a$ and  the $y$ intercept be $b$ ,Then
\begin{align}
a+b&=1\label{eq:11/10/4/31a}\\
ab&=-6 \label{eq:11/10/4/32a}
\\
\implies  a = 3, b = -2
\end{align}
Thus, the possible 
intercepts are
\begin{align}
\myvec{3\\0}, \myvec{0\\-2},
\myvec{-2\\0}, \myvec{0\\3}
\end{align}
yielding
\begin{align}		
\vec{m}=\myvec{3\\2} \text{or,} \myvec{-2\\3}
\end{align}
\begin{enumerate}
\item For 
\begin{align}
\vec{n}
=\myvec{-2 \\3} 
\end{align}
the equation of the line is 
\begin{align}
	\vec{n}^\top\brak{\vec{x}-\vec{A}} &= 0 \\
	\myvec { -2 & 3 } \vec{x}  &= 6  
\end{align}
\item  For 
\begin{align}
\vec{n}
=\myvec{-3 \\-2} 
\end{align}
the equation of the line is 
\begin{align}
    \vec{n}^\top\brak{\vec{x}-\vec{B}} &= 0 \\  
	\myvec { -3 & -2 }  \vec{x}  &= 6        
\end{align}
\end{enumerate}
