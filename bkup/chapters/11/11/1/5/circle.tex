\section{Excercise 11.1}
Find the equation of the circle with centre $(-a,-b)$ and radius $\sqrt{a^2-b^2}$.

\section{SOLUTION}
\fi
Since
\begin{align}
	\vec{c} &= \myvec{-a\\-b} \text{ and } r = \sqrt{a^2-b^2}
	\\
	\vec{u} &= \myvec{a\\b},\,
	f = \norm{\vec{u}}^2 - r^2
	  =2b^2
\end{align}
Thus, the equation of circle is 
\begin{align}
	\norm{\vec{x}}^2 +2 \myvec{a&b}\vec{x}+2b^2 &= 0       		       
\end{align}
