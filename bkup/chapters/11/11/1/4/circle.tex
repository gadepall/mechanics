Q4.Find the equation of the circle with centre $(1,1)$ and radius $\sqrt{2}$.

\solution
\fi
Given
\begin{align}
	\vec{c} &= \myvec{1\\1} \text{ and } r = \sqrt{2},
	\\
	\vec{u}&=\vec{-c}
	 = \myvec{-1\\-1}\\
	 \\
	f &= \norm{\vec{u}}^2 - r^2
	  =0	
\end{align}
Thus, the equation of circle is 
\begin{align}
	\norm{\vec{x}}^2 -2\myvec{1&1}\vec{x} = 0       		       
\end{align}
